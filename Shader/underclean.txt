float map(vec3 v)
{
	float index = voxel(v);
	
	float zContPosi = voxel(vec3(v.x, v.y, v.z + 1.0));
	
	if (index != 0.0)
	{
		if (zContPosi != 0.0)
		{
			return 0.0;
		}
	}
	
	return index;
}
