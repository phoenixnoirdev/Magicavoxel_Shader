// Shader qui permet de retirez tout les block a l'exeption de block de couleur selectionner
//
//
//
//Nouveautés
//
//Paramètres renommés pour plus de clarté (colotarget)
//Shader utilisable dans l'interface MagicaVoxel
//
//
//
//	colotarget	→ couleur ciblée
//



// xs_begin
// arg : { var = 'colotarget'	name = 'ColorTar'	value = '0' range = '1 255' step = '1' precision = '0' }



float map(vec3 v)
{
	float index = voxel(v);
	
	if (index != 0.0)
	{
		if (index != colotarget)
		{
				return 0.0;
		}
	}
	
	return index;
}
